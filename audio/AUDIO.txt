contructor audBackground

Description
	This contructs a background source

	audBackground(
		int buffersize
	);

Parameters

	buffersize	The size of the buffer queue, more buffers means more memory

Description
	This cleans up the background class

	~audBackground();

audLoadDir

Description
	This loads the files from the directory

	int audLoadDir(
		string folder,
		string type = ""
	);

Parameters

	folder		The folder with the files to load
	type		The type of file to load, "" loads all files in folder

Returns

	int		0 if successful, others if unsuccessful (error returns not determined)

audUpdateQueue

Description
	This removes played files from the queue, loads the next files into the buffer, then adds it to the queue to keep the queue at the buffersize

	int audUpdateQueue(
		int play
	);

Parameters

	play		decides if the queue will be restarted if it is stoped, 1 for yes and 0 for no

Returns

	int		0 if successful, others if unsuccessful (error returns not determined)
